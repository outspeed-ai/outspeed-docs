```tsx AudioContainer.tsx
import { Track } from "@outspeed/core";
import { RealtimeAudio } from "@outspeed/react";

type TAudioContainerProps = {
  audioTrack?: Track;
}

export function AudioContainer(props: TAudioContainerProps) {
  const { audioTrack } = props

  return (
    <div>
      <RealtimeAudio track={audioTrack} />
    </div>
  );
}
```
</RequestExample>

<Note>All of our components are built using Tailwind CSS. Make sure to set it up correctly by following the guide available [here](https://tailwindcss.com/docs/installation).</Note>


`RealtimeAudio` can manage local or remote audio tracks. It uses an HTML audio element internally to
play the audio track. 

## Props

<ParamField body="track" type="Track">
  A local or remote audio track.
</ParamField>

<ParamField body="volume" type="number">
  The volume level for the audio track, ranging from 0 to 100. The default is 100.
</ParamField>
