```tsx AudioVisualizerContainer.tsx
import { Track } from "@outspeed/core";
import { RealtimeAudioVisualizer } from "@outspeed/react";

type TAudioVisualizerContainerProps = {
  audioTrack?: Track;
}

export function AudioVisualizerContainer(props: TAudioVisualizerContainerProps) {
  const { audioTrack } = props

  return (
    <div>
      <RealtimeAudioVisualizer track={audioTrack} />
    </div>
  );
}
```
