    Here are some examples of events you can listen for and actions you can control.
    ```js
      realtime.on('speech-start', () => {
        console.log('Speech has started');
      });

      realtime.on('speech-end', () => {
        console.log('Speech has ended');
      });

      realtime.on('call-start', () => {
        console.log('Call has started');
      });

      realtime.on('call-end', () => {
        console.log('Call has stopped');
      });

      realtime.on('volume-level', (volume) => {
        console.log(`Assistant volume level: ${volume}`);
      });

      realtime.on('message', (message) => {
        console.log(message);
      });

      realtime.on('error', (e) => {
        console.error(e)
      });
    ```
