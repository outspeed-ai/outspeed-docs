---
title: "useCreateConfig"
description: "A hook that helps you build configuration for `useWebRTC`."
---

<RequestExample>
```ts useCreateConfig.ts
import { useCreateConfig } from "@outspeed/react"

const {
  functionURL,
  setFunctionURL,
  audioDeviceId,
  setAudioDeviceId,
  videoDeviceId,
  setVideoDeviceId,
  audioConstraints,
  setAudioConstraints,
  videoConstraints,
  setVideoConstraints,
  screenConstraints,
  setScreenConstraints,
  audioCodec,
  setAudioCodec,
  videoCodec,
  setVideoCodec,
  dataChannelOptions,
  setDataChannelOptions,
  rtcConfig,
  setRTCConfig,
  logger,
  setLogger,
  getConfig,
} = useCreateConfig()

```

</RequestExample>

It also offers a set of setters and getters to update or retrieve values.

### Returns

<ParamField body="functionURL" type="string">
- The URL of the function for the backend, obtained after deploying the app using `outspeed-client`.
</ParamField>

<ParamField body="setFunctionURL" type="(functionURL: string) => void">
- A function to update the `functionURL`.
</ParamField>

<ParamField body="audioDeviceId" type="string | null">
- The ID of the selected audio device. `null` if no device is selected.
</ParamField>

<ParamField body="setAudioDeviceId" type="(audioDeviceId: string | null) => void">
- A function to update the `audioDeviceId`.
</ParamField>

<ParamField body="videoDeviceId" type="string | null">
- The ID of the selected video device. `null` if no device is selected.
</ParamField>

<ParamField body="setVideoDeviceId" type="(videoDeviceId: string | null) => void">
- A function to update the `videoDeviceId`.
</ParamField>

<ParamField body="audioConstraints" type="MediaTrackConstraints">
- The audio constraints used for obtaining media from the `navigator`.
</ParamField>

<ParamField body="setAudioConstraints" type="(audioConstraints: MediaTrackConstraints) => void">
- A function to update the `audioConstraints`.
</ParamField>

<ParamField body="videoConstraints" type="MediaTrackConstraints">
- The video constraints used for obtaining media from the `navigator`.
</ParamField>

<ParamField body="setVideoConstraints" type="(videoConstraints: MediaTrackConstraints) => void">
- A function to update the `videoConstraints`.
</ParamField>

<ParamField body="screenConstraints" type="MediaTrackConstraints">
- The screen constraints used for capturing screen media.
</ParamField>

<ParamField body="setScreenConstraints" type="(screenConstraints: MediaTrackConstraints) => void">
- A function to update the `screenConstraints`.
</ParamField>

<ParamField body="audioCodec" type="string">
- The codec used for audio.
</ParamField>

<ParamField body="setAudioCodec" type="(audioCodec: string) => void">
- A function to update the `audioCodec`.
</ParamField>

<ParamField body="videoCodec" type="string">
- The codec used for video.
</ParamField>

<ParamField body="setVideoCodec" type="(videoCodec: string) => void">
- A function to update the `videoCodec`.
</ParamField>

<ParamField body="dataChannelOptions" type="RTCDataChannelInit">
- Options for configuring the WebRTC data channel.
</ParamField>

<ParamField body="setDataChannelOptions" type="(dataChannelOptions: RTCDataChannelInit) => void">
- A function to update the `dataChannelOptions`.
</ParamField>

<ParamField body="rtcConfig" type="RTCConfiguration">
- The configuration used for setting up the WebRTC peer connection.
</ParamField>

<ParamField body="setRTCConfig" type="(rtcConfig: RTCConfiguration) => void">
- A function to update the `rtcConfig`.
</ParamField>

<ParamField body="logger" type="Logger">
- The logger instance used for handling and storing logs.
</ParamField>

<ParamField body="setLogger" type="(logger: Logger) => void">
- A function to update the `logger`.
</ParamField>

<ParamField body="getConfig" type="() => TRealtimeWebSocketConfig">
- A function that builds and returns the configuration object.
- Internally, it passes all inputs to [createConfig](./createConfig) and returns its output.
</ParamField>
